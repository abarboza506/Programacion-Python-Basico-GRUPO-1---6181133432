    datos_pacientes = json.load(archivo)

# Crear un diccionario para contar pacientes por provincia

    # Si la provincia ya est√° en el diccionario, sumar 1
    if provincia in conteo_provincias:

    else:

# Encontrar la provincia con el m√°ximo de pacientes
